# Git Setup Commands

# Step 1: Initialize Git repository
git init

# Step 2: Add all files to staging area
git add .

# Step 3: Create initial commit
git commit -m "Initial commit: Full-stack web app with Next.js and Node.js"

# Step 4: Add remote repository (replace with your repository URL)
git remote add origin https://github.com/yourusername/fullstack-web-app.git

# Step 5: Push to repository
git push -u origin main

# Alternative if your default branch is 'master'
# git push -u origin master
